25      private static final String HEADER_4 = "firstname";
26      private static final String HEADER_5 = "lastname";
27      private static final String HEADER_6 = "password";
28      private static final String HEADER_7 = "id";
29  
30      @Test
31      public void testReadExcel() throws Exception {
32          try (InputStream input = getClass().getResourceAsStream("/excel/testMap.xlsx")) {
33              List<Map<String, String>> maps = new ExcelMapReader().read(input);
34              assertNotNull(maps);
35              assertEquals(488, maps.size());
36              for (Map<String, String> map : maps) {
37                  assertEquals(7, map.size());
38                  assertNotNull(map.get(HEADER_1));
39                  assertNotNull(map.get(HEADER_2));
40                  assertNotNull(map.get(HEADER_3));
41                  assertNotNull(map.get(HEADER_4));
42                  assertNotNull(map.get(HEADER_5));
43                  assertNotNull(map.get(HEADER_6));
44                  assertNotNull(map.get(HEADER_7));
45              }
46          }
47      }
48  
49      @Test
50      public void testReadExcelWithMapping() throws Exception {
51          String mapping1 = "mapping1";
52          String mapping2 = "mapping2";
53          List<String> passwords = new ArrayList<>();
54          passwords.add("customer1");
55          passwords.add("customer2");
56  
57          try (InputStream input = getClass().getResourceAsStream("/excel/testMap.xlsx")) {
58              Map<String, String> mapping = new HashMap<>();
59              mapping.put("password", mapping1);
60              mapping.put("id", mapping2);
61              List<Map<String, String>> maps = new ExcelMapReader().read(input, mapping);
62              assertNotNull(maps);
63              assertEquals(488, maps.size());
64              for (Map<String, String> map : maps) {
65                  assertEquals(2, map.size());
66                  assertThat(passwords, hasItem(map.get(mapping1)));
67                  assertThat(Double.valueOf(map.get(mapping2)).intValue(), allOf(greaterThan(0), lessThan(489)));
68              }
69          }
70      }
