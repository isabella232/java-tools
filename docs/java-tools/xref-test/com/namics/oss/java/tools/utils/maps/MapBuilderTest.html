5   package com.namics.oss.java.tools.utils.maps;

7   import org.junit.Test;
8   
9   import java.util.Map;
10  import java.util.SortedMap;
11  import java.util.TreeMap;
12  
13  import static org.junit.Assert.assertEquals;
14  import static org.junit.Assert.assertNotNull;
15  import static org.junit.Assert.assertTrue;
16  
17  /**
18   * MapBuilderTest.
19   *
20   * @author aschaefer, Namics AG
21   * @since 13.06.14 11:00
22   */
23  public class MapBuilderTest {
24  
25  	@Test
26  	public void testSpecificMap(){
27  		assertTrue(new MapBuilder<String,String>(new TreeMap<String,String>().getClass()).map() instanceof TreeMap);
28  		assertTrue(new MapBuilder<String,String>(TreeMap.class).map() instanceof TreeMap);
29  		assertTrue(new MapBuilder<String,String>(new TreeMap<String,String>()).map() instanceof TreeMap);
30  
31  		TreeMap<String, String> instances = new TreeMap<String, String>();
32  		assertTrue(instances == new MapBuilder<String, String>(instances).map());
33  	}
34  
35  	@Test
36  	public void testPut() throws Exception {
37  		MapBuilder<String, String> test = new MapBuilder<String, String>()
38  				.put("Test", "Test")
39  				.put("Test2", "Test");
40  		Map<String, String> map = test.map();
41  		assertNotNull(map);
42  		assertEquals(2, map.size());
43  	}
44  
45  	@Test
46  	public void testMap() throws Exception {
47  		MapBuilder<String, String> test = new MapBuilder<String, String>()
48  				.put("Test", "Test")
49  				.put("Test2", "Test");
50  		Map<String, String> map = test.map();
51  		assertNotNull(map);
52  		assertEquals(2, map.size());
53  	}
54  
55  	@Test(expected = UnsupportedOperationException.class)
56  	public void testImmutable() throws Exception {
57  		Map<String, String> map = new MapBuilder<String, String>()
58  				.put("KEY", "VALUE")
59  				.put("ANOTHER", "VALUE")
60  				.immutable();
61  		assertNotNull(map);
62  		assertEquals(2, map.size());
63  		map.put("Immutable", "cannot add");
64  	}
65  
66  	@Test
67  	public void testImmutableUnderlyingMap() throws Exception {
68  		MapBuilder<String, String> builder = new MapBuilder<String, String>();
69  		Map<String, String> map = builder
70  				.put("KEY", "VALUE")
71  				.put("ANOTHER", "VALUE")
72  				.immutable();
73  		builder.put("test","test");
74  		assertEquals(2,map.size());
75  		assertEquals(3,builder.immutable().size());
76  	}
77  
78  	@Test
79  	public void testSorted() throws Exception {
80  		MapBuilder<String, String> test = new MapBuilder<String, String>()
81  				.put("Test", "Test")
82  				.put("Test2", "Test");
83  		Map<String, String> map = test.sorted();
84  		assertTrue(map instanceof SortedMap);
85  		assertEquals(2, map.size());
86  
87  		TreeMap<String, String> instances = new TreeMap<String, String>();
88  		assertTrue(instances == new MapBuilder<String, String>(instances).sorted());
89  	}
90  }
