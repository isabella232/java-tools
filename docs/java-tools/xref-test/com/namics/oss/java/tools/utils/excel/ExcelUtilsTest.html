25   * ExcelUtilsTest.

30  public class ExcelUtilsTest {
31  	private static final Logger LOG = LoggerFactory.getLogger(ExcelUtilsTest.class);
32  
33  	TestBean[] testBeans = new TestBean[] {
34  			new TestBean().username("hmuster").firstname("Hans").lastname("Muster").id("1").pleaseIgnore("ignoreProp"),
35  			new TestBean().username("emuster").firstname("Erika").lastname("Muster").id("2").pleaseIgnore("ignoreProp"),
36  	};
37  
38  
39  	@Test
40  	public void testWriteRead() throws Exception {
41  		String absolute = getClass().getResource("/").getFile() + "excel/util-test.xlsx";
42  		LOG.info("{}", absolute);
43  		try (OutputStream out = new FileOutputStream(absolute)) {
44  			ExcelUtils.write(Arrays.asList(testBeans), out, "pleaseIgnore");
45  		}
46  
47  		try (InputStream input = new FileInputStream(absolute)) {
48  			List<TestBean> verify = ExcelUtils.read(TestBean.class, input);
49  			for (TestBean testBean : verify) {
50  				LOG.info("{}", testBean);
51  			}
52  			assertThat(verify, contains(sameBeanAs(testBeans[0]).ignoring("pleaseIgnore").with("pleaseIgnore", isEmptyOrNullString()),
53  			                            sameBeanAs(testBeans[1]).ignoring("pleaseIgnore").with("pleaseIgnore", isEmptyOrNullString())));
54  		}
55  	}
56  }
