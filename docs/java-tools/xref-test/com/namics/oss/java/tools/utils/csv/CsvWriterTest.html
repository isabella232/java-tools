10  import static org.hamcrest.text.IsEqualIgnoringWhiteSpace.equalToIgnoringWhiteSpace;
11  import static org.junit.Assert.assertThat;

13  /**
14   * CsvWriterTest.

16   * @author lboesch, Namics AG

19  public class CsvWriterTest {
20  
21  	TestBean[] testBeans = new TestBean[] {
22  			new TestBean().username("hmuster").firstname("Hans").lastname("Muster").id("1").pleaseIgnore("ignoreProp"),
23  			new TestBean().username("emuster").firstname("Erika").lastname("Muster").id("2").pleaseIgnore("ignoreProp"),
24  	};
25  
26  	CsvWriter service = new CsvWriter();
27  
28  	@Test
29  	public void testWrite() throws IOException {
30  		ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
31  		service.write(Arrays.asList(testBeans), outputStream, null, "pleaseIgnore");
32  		assertThat(outputStream.toString(), equalToIgnoringWhiteSpace("firstname,id,lastname,username "
33  		                                                              + "Hans,1,Muster,hmuster "
34  		                                                              + "Erika,2,Muster,emuster"));
35  
36  	}
37  
38  	@Test
39  	public void testWriteWithHeaders() throws IOException {
40  		ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
41  		service.writeWithHeaders(Arrays.asList(testBeans), outputStream, null, "username", "lastname");
42  		assertThat(outputStream.toString(), equalToIgnoringWhiteSpace("username,lastname "
43  		                                                              + "hmuster,Muster "
44  		                                                              + "emuster,Muster"));
45  
46  	}
47  
48  	@Test
49  	public void testWriteWithHeadersSeparator() throws IOException {
50  		ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
51  		int delimiter = ';';
52  		service.writeWithHeaders(Arrays.asList(testBeans), outputStream, delimiter, "username", "lastname");
53  		assertThat(outputStream.toString(), equalToIgnoringWhiteSpace("username;lastname "
54  		                                                              + "hmuster;Muster "
55  		                                                              + "emuster;Muster"));
56  
57  	}
