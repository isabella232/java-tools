7   /**
8    * PropertiesUtils.
9    *
10   * @author aschaefer, Namics AG
11   * @since 27.08.15 14:08
12   */
13  public abstract class PropertiesUtils {
14  
15  	private PropertiesUtils() {
16  		throw new UnsupportedOperationException("PropertiesUtils cannot be instantiated");
17  	}
18  
19  	/**
20  	 * Create PropertiesBuilder and put a value in it, all PropertiesBuilder#properties to get actual Properties.
21  	 *
22  	 * @param key   property key to put
23  	 * @param value property value to put
24  	 * @return PropertiesBuilder instance with provided property, call PropertiesBuilder#properties for actual Properties.
25  	 */
26  	public static PropertiesBuilder put(String key, String value) {
27  		return new PropertiesBuilder().put(key, value);
28  	}
29  }
